Merge Rule: Higher priority layers override lower layers. If a field is not set in a higher layer, the value is inherited from the next layer down. This allows templates to override only specific fields while inheriting grove-wide defaults for everything else.

// Load research config from Supabase (FIXED) export async function loadResearchAgentConfig( groveId: string ): Promise<ResearchAgentConfigPayload> { const supabase = getSupabaseClient(); // Query active config from database const { data } = await supabase .from('research_agent_configs') .select('payload') .eq('grove_id', groveId) .eq("meta->>'status'", 'active') .single(); // Return loaded config or fall back to defaults if (data?.payload) { return data.payload; } console.warn('[ConfigLoader] No active research config, using defaults'); return DEFAULT_RESEARCH_AGENT_CONFIG_PAYLOAD; } // Load writer config from Supabase (FIXED) export async function loadWriterAgentConfig( groveId: string ): Promise<WriterAgentConfigPayload> { const supabase = getSupabaseClient(); const { data } = await supabase .from('writer_agent_configs') .select('payload') .eq('grove_id', groveId) .eq("meta->>'status'", 'active') .single(); if (data?.payload) { return data.payload; } console.warn('[ConfigLoader] No active writer config, using defaults'); return DEFAULT_WRITER_AGENT_CONFIG_PAYLOAD; } // Merge configs with template overrides export function mergeConfigs( writerConfig: WriterAgentConfigPayload, templateOverrides?: Partial<WriterAgentConfigPayload> ): WriterAgentConfigPayload { // Template overrides win over grove-wide config return { ...writerConfig, ...templateOverrides, voice: { ...writerConfig.voice, ...templateOverrides?.voice }, documentStructure: { ...writerConfig.documentStructure, ...templateOverrides?.documentStructure }, qualityRules: { ...writerConfig.qualityRules, ...templateOverrides?.qualityRules }, }; }
